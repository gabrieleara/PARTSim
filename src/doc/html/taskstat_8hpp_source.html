55  class PreemptionStat : public Measure {

57  Tick descTime;
58  Tick schedTime;
59  public:
60  PreemptionStat(string name = "") : Measure(name) {};
61 
62  void probe(const DeschedEvt &e) {
63  if (e.getLastTime() != schedTime) {
64  descTime = e.getLastTime();
65  count ++;
66  }
67  if (e.getLastTime() < Measure::_transitory)
68  count = 0;
69  }
70 
71  void probe(const SchedEvt &e) {
72  if (e.getLastTime() != descTime)
73  schedTime = e.getLastTime();
74  else count --;
75  if (e.getLastTime() < Measure::_transitory)
76  count = 0;
77  }
78 
79  void probe(const EndEvt &e) {
80  Measure::record(count);
81  count = 0;
82  }
